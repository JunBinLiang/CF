class Solution{
    PrintWriter out;
    int INF=Integer.MAX_VALUE;
    int MOD=1000000007;
    public Solution(PrintWriter out){
        this.out=out;
    }


    public void solution(int A[]){
        long res=0;
        int N=A.length;
        if(A[N-1]==-1){
            out.println(0);
            return;
        }

        PriorityQueue<Integer>pq=new PriorityQueue<>();

        res+=A[N-1];
        int cnt=0;
        int target=N/2;
        for(int i=N-2;i>=0;i--){
            if(A[i]==-1)break;
            cnt++;pq.add(A[i]);
            if(cnt==target){
                int mn=pq.poll();
                //System.out.println(mn+"         "+cnt);
                res+=mn;
                cnt=0;
                target/=2;
            }
        }
        out.println(res);
    }

}
