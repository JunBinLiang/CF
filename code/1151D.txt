class Solution{
    PrintWriter out;
    int INF=Integer.MAX_VALUE;
    int MOD=1000000007;
    public Solution(PrintWriter out){
        this.out=out;
    }

    public void solution(int A[][]) {
        int n=A.length;
        long res=0;
        //(0,3) (1,2) (2,1) (3,0)  => 4 pairs
        Arrays.sort(A,(a,b)->{
            return (b[0]-b[1])-(a[0]-a[1]);
        });

        for(int i=0;i<n;i++){
            long lcnt=i;
            long rcnt=n-i-1;
            long sum=lcnt*A[i][0]+rcnt*A[i][1];
            res+=sum;
        }
        out.println(res);
    }

}
