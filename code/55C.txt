class Solution{
    PrintWriter out;
    int INF=Integer.MAX_VALUE;
    int MOD=1000000007;
    public Solution(PrintWriter out){
        this.out=out;
    }

    long res=Long.MAX_VALUE;

    public void solution(int n,int m,int A[][]) {

        boolean res=false;
        for(int p[]:A){
            int x=p[0],y=p[1];
            if(x<=5||x>=n-4||y<=5||y>=m-4)res=true;
        }

        if(res)out.println("YES");
        else out.println("NO");
    }
