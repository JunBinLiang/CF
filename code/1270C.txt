    static class FastScanner {//scanner from SecondThread

        public String next() {
            while (!st.hasMoreElements())
                try {
                    st=new StringTokenizer(br.readLine());
                } catch (IOException e) {
                    e.printStackTrace();
                }
            return st.nextToken();
        }
        int Int() {
            return Integer.parseInt(next());
        }

        long Long() {
            return Long.parseLong(next());
        }

        String Str(){
            return next();
        }
    }

class Solution{
    long dp[][];
    public void solution(PrintWriter out,int A[]){
      long sum=0;
      long xor=0;
      for(int i:A){
          sum+=i;
          xor^=i;
      }

      if(sum==2*xor){
          out.println(0);
          return;
      }
      List<Long>res=new ArrayList<>();
      res.add(xor);
      sum+=xor;//xor==0
      res.add(sum);


      out.println(res.size());
      for(long i:res){
          out.print(i+" ");
      }
      out.println();

    }




}
