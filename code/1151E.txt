class Solution{
    PrintWriter out;
    int INF=Integer.MAX_VALUE;
    int MOD=1000000007;
    public Solution(PrintWriter out){
        this.out=out;
    }


    public void solution(int A[]) {
        //We leave in the tree only vertices whose values range from l to r.
        //The value of the function will be the number of connected components in the new graph.
        //f(l,r)

        int n=A.length;
        long res=0;
        res+=(A[0]+0l)*(n-A[0]+1);

        for(int i=1;i<A.length;i++){
            if(A[i]==A[i-1])continue;
            int mn=Math.min(A[i-1],A[i]);
            int mx=Math.max(A[i-1],A[i]);
            long cnt=(A[i]+0l)*(n-A[i]+1);
            long extra=(mn+0l)*(n-mx+1);
            res+=(cnt-extra);
        }
        out.println(res);
    }







}
