class Solution{
    public void solution(PrintWriter out,int A[]){
        long res=0;
        int n=A.length;
        int dp[]=new int[n];

        for(int i=0;i<A.length;i++){
            if(A[i]<=1){
                if(i+1<A.length){
                    dp[i+1]+=dp[i];
                }
                continue;
            }

            for(int j=2;j<=A[i];j++){
                if(i+j>=A.length)break;
                dp[i+j]++;
            }

            if(A[i]>dp[i]){
                A[i]-=dp[i];
                res+=(A[i]-1);
            }
            else{
                dp[i]=dp[i]-(A[i]-1);
                if(i+1<A.length){
                    dp[i+1]+=dp[i];
                }
                A[i]=1;
            }
        }


        System.out.println(res);
    }

    public int dfs(int dp[],int A[],int i){
        if(i>=A.length){
            return 0;
        }
        if(dp[i]!=-1){
            return dp[i];
        }
        int res=1+dfs(dp,A,i+A[i]);
        dp[i]=res;
        return res;
    }

    public void go(int A[],int i){
        if(i>=A.length){
            return;
        }
        go(A,i+A[i]);
        A[i]=Math.max(1,A[i]-1);
    }

    public void print(int A[]){
        for(int i:A){
            System.out.print(i+" ");
        }
        System.out.println();
        System.out.println();
    }

}
